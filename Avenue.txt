function copyToClipboard(text) {
  var tempTextarea = document.createElement('textarea');
  tempTextarea.style.position = 'fixed';
  tempTextarea.style.opacity = 0;

  // Replace line breaks with actual line breaks
  var textWithLineBreaks = text.replace(/<br\s*[\/]?>/gi, '\n');

  tempTextarea.value = textWithLineBreaks;

  document.body.appendChild(tempTextarea);

  tempTextarea.select();
  document.execCommand('copy');

  document.body.removeChild(tempTextarea);

  notify('success', 'Successful');
}

     <script>
        let futureDay = '';

        function getCurrentDateInPKT() {
            const now = new Date();
            const utcTime = now.getTime() + (now.getTimezoneOffset() * 60000);
            const PKTOffset = 5 * 60 * 60000; // PKT is UTC+5
            return new Date(utcTime + PKTOffset);
        }

        function initializeFutureDay() {
            const options = { weekday: 'long' };

            try {
                const currentDate = getCurrentDateInPKT();

                // Calculate the future date based on PKT
                let futureDate = new Date(currentDate);
                if (currentDate.getHours() >= 8) {
                    futureDate.setDate(futureDate.getDate() + 2);
                } else {
                    futureDate.setDate(futureDate.getDate() + 1);
                }
                futureDate.setHours(8, 0, 0, 0); // Set future date time to 8 AM PKT

                futureDay = futureDate.toLocaleDateString('en-US', options);

                // Optionally update any display here
                // document.getElementById('futureDayDisplay').innerText = 'Text before: ' + futureDay + ' :Text after';
            } catch (error) {
                console.error('Error calculating future date:', error);
            }
        }

        function copyText(preText, postText) {
            if (futureDay === '') {
                alert('Future day is not calculated yet. Please wait.');
                return;
            }

            const combinedText = preText + futureDay + postText;

            navigator.clipboard.writeText(combinedText).then(() => {
                console.log('Text copied to clipboard');
                showNotification();
            }).catch(err => {
                console.error('Could not copy text: ', err);
            });
       



</script>
